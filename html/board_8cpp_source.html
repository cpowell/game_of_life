18 int Board::getLiveNeighborCountForCell(const int x, const int y) const {
19  int above = y-1;
20  if (above < 0)
21  above = BOARD_HEIGHT;
22 
23  int below = y+1;
24  if (below >= BOARD_HEIGHT)
25  below = 0;
26 
27  int left = x-1;
28  if (left < 0)
29  left = BOARD_WIDTH;
30 
31  int right = x+1;
32  if (right >= BOARD_WIDTH) {
33  right = 0;
34  }
35 
36  int live = 0;
37  for (int xx=left; xx<=right; ++xx) {
38  for (int yy=above; yy<=below; ++yy) {
39  if (xx==x && yy==y)
40  continue;
41 
42  live += _cells[xx][yy];
43  }
44  }
45  return live;
46 }
